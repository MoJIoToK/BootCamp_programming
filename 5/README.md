# Занятие №5

На этом уроке разберем сортировку пузырьком

Сложность всегда n^2. Так как цикл вложен в цикл.

Работа парами.

## Пример
Сортировка пузырьком
[3, 1, 5, 0, 7, 9, 8] - начальный массив

1. [1, 3, 5, 0, 7, 9, 8]
2. [1, 3, 0, 5, 7, 9, 8]
3. [1, 0, 3, 5, 7, 9, 8]
4. [0, 1, 3, 5, 7, 9, 8]
5. [0, 1, 3, 5, 7, 8, 9]
6. [0, 1, 3, 5, 7, 8, 9]
7. [0, 1, 3, 5, 7, 8, 9]
